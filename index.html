async function loadImages(refresh = false) { if (refresh) pageToken = ''; if (isLoading) return; isLoading = true; try { const data = await apiCos('list', { marker: pageToken, maxKeys: pageSize }); // 兼容新 API 格式 (data.files) 和旧格式 (data.Contents) let images = []; if (data.files && Array.isArray(data.files)) { // 新格式：API 直接返回 files 数组 images = data.files.map(item => ({ key: item.key, name: item.key.split('/').pop(), size: parseInt(item.size) || 0, lastModified: item.lastModified, url: item.url })); pageToken = ''; } else if (data.Contents) { // 旧格式：COS SDK 原生返回 images = (data.Contents || []).map(item => ({ key: item.Key, name: item.Key.split('/').pop(), size: parseInt(item.Size) || 0, lastModified: item.LastModified, url: `${CONFIG.domain}/${item.Key}` })); pageToken = data.NextMarker || ''; } const search = document.getElementById('searchInput').value.toLowerCase().trim(); if (search) images = images.filter(img => img.name.toLowerCase().includes(search)); const sort = document.getElementById('sortSelect').value; images.sort((a, b) => { const da = new Date(a.lastModified), db = new Date(b.lastModified); return sort === 'newest' ? db - da : da - db; }); if (refresh) { currentImages = images; } else { currentImages = currentImages.concat(images); } renderGallery(); document.getElementById('totalCount').textContent = currentImages.length + ' 张截图'; document.getElementById('loadMoreBtn').style.display = 'none'; } catch (err) { console.error(err); showToast('加载失败', 'error'); } finally { isLoading = false; } }
